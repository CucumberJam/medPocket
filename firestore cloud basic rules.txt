service cloud.firestore {
  match /databases/users/documents {
  
    match /analyses/documents {
      allow read, write: if request.auth != null;
        match /analysis_list/{analysis} {
          allow read, write: if request.auth != null;
        }
    }
    
    match /medications/{medication}{
      allow read, write: if request.auth != null;
    }
  }
}


// для чтения и записи только одному пользователю, а путь к данным содержит идентификатор пользователя:

service cloud.firestore {
  match /databases/users/documents {
    match /{userId}/analyses/documents {
			match /{analysesId}/analysis_list/{documents=**} {
           allow read, write, update, delete: if request.auth != null && request.auth.uid == userId
         }
    }
    match /{userId}/medications/{documents=**} {
      allow read, write, update, delete: if request.auth != null && request.auth.uid == userId
    }
  }
}
